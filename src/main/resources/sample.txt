object Solution {\n\n  def maximumPerimeterTriangle(sticks: Array[Int]): Option[Array[Int]] = {\n    sticks.sorted(Ordering[Int].reverse).sliding(3, 1).find(arr => arr.tail.sum > arr.head)\n  }\n\n  def main(args: Array[String]) {\n    val stdin = scala.io.StdIn\n    val n = stdin.readLine.trim.toInt\n    val sticks = stdin.readLine.split(\" \").map(_.trim.toInt)\n\n    maximumPerimeterTriangle(sticks) match {\n      case Some(result) => println(result.reverse.mkString(\" \"))\n      case _ => println(-1)\n    }\n  }\n}\n
